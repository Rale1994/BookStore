import java.util.Arrays;

public class main {
    public static void main(String[] args) {
        int[] test = new int[]{1, 1, 2, 2, 3, 4, 5};
        System.out.println("Arrays without duplicate: " + Arrays.toString(vratiDuplikate(test)));

    }

    public static int[] vratiDuplikate(int[] niz) {
        int[] novNiz = new int[niz.length];
        int prethodni = niz[0];
        novNiz[0]=prethodni;
        for (int i = 1; i < niz.length; i++) {
            int trenutni = niz[i];
            if (prethodni != trenutni) {
                novNiz[i] = trenutni;
            }
            prethodni = trenutni;
        }
        return novNiz;
    }
}
